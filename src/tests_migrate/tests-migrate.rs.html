use lazy_static::lazy_static;

mod data;
mod settings;

pub use data::Data;
pub use settings::Settings;

lazy_static! {
    pub static ref SETTINGS: Settings = Settings::new().unwrap();
}

#[actix_web::main]
async fn main() {
    let data = Data::new().await;
    pretty_env_logger::init();

    sqlx::migrate!("./migrations/").run(&data.db).await.unwrap();
}
